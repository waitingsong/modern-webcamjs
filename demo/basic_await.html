Modern WebcamJS Test Page async/await

Demonstrates 320x240 capture & display

captured image will appear here...

    
  <script src="./require.min.js" ></script>
  <script>
    requirejs(['../dist/webcam'], ({webcam}) => {
        // create an instance and initialize it
        const config = {
            ctx: '#my_camera',
            previewWidth: 320,
            previewHeight: 240,
            imageFormat:  'jpeg',
            jpegQuality:  97,
        }

        webcam(config).connect().then(cam => {
            window.cam = cam;
        });
    });

    async function take_snapshot() {
        const url = await cam.snap();

        console.log('url len:' + url.length);
        document.querySelector('img#snap').src = url;
    }

  </script>

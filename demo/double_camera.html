  <script src="./require.min.js" ></script>
  <script>
    requirejs(['../dist/webcam'], ({Webcam}) => {
        // create an instance and initialize it
        const config = {
            ctx: '#my_camera',
            previewWidth: 400,
            previewHeight: 300,
            imageFormat:  'jpeg',
            jpegQuality:  97,
        }
        const cam = Webcam(config);
        // multi snap options for multi device
        const opts = [
            {
                streamIdx: 0,   // start at 0
                width:    400,
                height:   300,
            },
            {
                streamIdx: 1,
                width:    320,
                height:   240,
                jpegQuality:  95,
            },
        ];
        cam.set(opts).connect(0);   // preview camera index 0

        window.cam = cam;
    });

    function take_snapshot(sidx) {
        cam.snap(sidx).then(url => {
          console.log('url len:' + url.length);
          document.querySelector('img#snap').src = url;
        });
    }

    function toggle_camera() {
        if (cam.currStreamIdx === 0) {
            cam.connect(1);
        }
        else {
            cam.connect(0);
        }
    }

  </script>

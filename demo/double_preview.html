Modern WebcamJS Test Page

Demonstrates double cameras with double preview

    
  <script src="./require.min.js" ></script>
  <script>
    requirejs(['../dist/webcam'], ({Webcam}) => {
        // create an instance and initialize it
        const config0 = {
            ctx: '#my_camera_0',
            streamIdx: 0,   // start at 0
        }

        Webcam(config0).connect(0); 

        // create another instance and initialize it
        const config1 = {
            ctx: '#my_camera_1',
            streamIdx: 1,
        }

        Webcam(config1).connect(1);
    });

  </script>

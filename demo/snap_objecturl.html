Modern WebcamJS Test Page

Demonstrates 400x300 display & capture return objectURL

captured image will appear here...

    
  <script src="./require.min.js" ></script>
  <script>
    requirejs(['../dist/webcam'], ({Webcam}) => {
        // create an instance and initialize it
        const config = {
            ctx: '#my_camera',
            previewWidth: 400,
            previewHeight: 300,
            imageFormat:  'jpeg',
            jpegQuality:  97,
        }

        Webcam(config).connect().then(cam => {
            window.cam = cam;
        });
    });

    function take_snapshot() {
        const opts = {
            dataType: 'objectURL',  // default is dataURL
        };

        cam.snap(opts).then(url => { 
            // url is objectURL, need destroy it via URL.revokeObjectURL(url) when unusded no more
            console.log('url len:' + url.length);
            document.querySelector('img#snap').src = url;
        });
    }
  </script>
